#### Hierarchical Queries

<p align="center">

    SELECT 
        emp_id,
        emp_first_name,
        CONNECT_BY_ROOT emp_first_name as MANAGER_NAME,
        LEVEL
    FROM employee
    WHERE LEVEL > 1 
    CONNECT BY PRIOR emp_id = manager_id
    ORDER SIBLINGS BY emp_id;

.

    SELECT 
        emp_id,
        LEVEL,
        SYS_CONNECT_BY_PATH(emp_first_name, '/') "Path"
    FROM employee
    CONNECT BY PRIOR emp_id = manager_id
    ORDER SIBLINGS BY emp_id;
